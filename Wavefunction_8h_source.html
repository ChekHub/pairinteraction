81 template < typename T >
82 real_t IntegrateRadialElement( QuantumDefect const& qd1, int power, QuantumDefect const& qd2)
83 {
84  T N1(qd1);
85  T N2(qd2);
86 
87  auto const& x1 = N1.axis();
88  auto const& y1 = N1.integrate();
89  auto const& x2 = N2.axis();
90  auto const& y2 = N2.integrate();
91  auto const dx = N1.dx;
92 
93  auto const xmin = x1.front() >= x2.front() ? x1.front() : x2.front();
94  auto const xmax = x1.back() <= x2.back() ? x1.back() : x2.back();
95 
96  real_t mu = 0;
97  // If there is an overlap, calculate the matrix element
98  if (xmin <= xmax) {
99  int start1 = findidx(x1, xmin);
100  int end1 = findidx(x1, xmax);
101  int start2 = findidx(x2, xmin);
102  int end2 = findidx(x2, xmax);
103 
104  int i1, i2;
105  for (i1 = start1, i2 = start2; i1 < end1 && i2 < end2; ++i1, ++i2)
106  {
107  mu += y1[i1]*y2[i2] * std::pow(x1[i1], T::power_kernel(power)) * dx;
108  }
109  mu = std::abs(2*mu);
110  }
111 
112  return mu;
113 }
